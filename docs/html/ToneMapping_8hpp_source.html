39 void apply(const double* framebuffer, unsigned char* output, int pixel_count, int channels) const {
40 for(int base_index = 0; base_index < pixel_count * channels; base_index += channels) {
41 switch(channels) {
42 case 1:
43 output[base_index] = static_cast<unsigned char>(convertToLDR(framebuffer[base_index]) * NORMALIZED_TO_COLOR8);
44 break;
45 case 3:
46 case 4: {
47 const ColorRGB pixel_color = {framebuffer[base_index], framebuffer[base_index + 1],
48 framebuffer[base_index + 2]};
49 ColorRGB ldr_color = convertToLDR(pixel_color);
50 ldr_color.clamp(0.0, 1.0);
51
52 output[base_index] = static_cast<unsigned char>(ldr_color.r * NORMALIZED_TO_COLOR8);
53 output[base_index + 1] = static_cast<unsigned char>(ldr_color.g * NORMALIZED_TO_COLOR8);
54 output[base_index + 2] = static_cast<unsigned char>(ldr_color.b * NORMALIZED_TO_COLOR8);
55 break;
56 }
57 default:
58 throw std::runtime_error("Unsupported number of channels for tone mapping: " + std::to_string(channels));
59 }
60
61 if(channels == 4) {
62 const double alpha = std::clamp(framebuffer[base_index + 3], 0.0, 1.0);
63 output[base_index + 3] = static_cast<unsigned char>(alpha * NORMALIZED_TO_COLOR8);
64 }
65 }
66 }
