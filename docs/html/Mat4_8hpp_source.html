318 static Mat4 LookAt(const Vec3<T>& eye, const Vec3<T>& center, const Vec3<T>& up) {
319 const Vec3<T> forward = (center - eye).normalized();
320 const Vec3<T> side = forward.cross(up).normalized();
321 const Vec3<T> upVector = side.cross(forward);
322
323 Mat4 lookAtMatrix;
324 lookAtMatrix.m[0][0] = side.x;
325 lookAtMatrix.m[0][1] = side.y;
326 lookAtMatrix.m[0][2] = side.z;
327 lookAtMatrix.m[0][3] = -(side.x * eye.x + side.y * eye.y + side.z * eye.z);
328
329 lookAtMatrix.m[1][0] = upVector.x;
330 lookAtMatrix.m[1][1] = upVector.y;
331 lookAtMatrix.m[1][2] = upVector.z;
332 lookAtMatrix.m[1][3] = -(upVector.x * eye.x + upVector.y * eye.y + upVector.z * eye.z);
333
334 lookAtMatrix.m[2][0] = -forward.x;
335 lookAtMatrix.m[2][1] = -forward.y;
336 lookAtMatrix.m[2][2] = -forward.z;
337 lookAtMatrix.m[2][3] = (forward.x * eye.x + forward.y * eye.y + forward.z * eye.z);
338
339 lookAtMatrix.m[3][0] = 0.0;
340 lookAtMatrix.m[3][1] = 0.0;
341 lookAtMatrix.m[3][2] = 0.0;
342 lookAtMatrix.m[3][3] = 1.0;
343
344 return lookAtMatrix;
345 }

373 static Mat4 Orthographic(T left, T right, T bottom, T top, T near, T far) {
374 Mat4 ortho;
375 ortho.m[0][0] = 2 / (right - left);
376 ortho.m[1][1] = 2 / (top - bottom);

381 ortho.m[2][3] = -(far + near) / (far - near);

395 static Mat4 Perspective(T fovY, T aspect, T near, T far) {
396 T invTanHalfFovY = 1.0 / std::tan(fovY * HALF);
397 Mat4 persp;
398 persp.m[0][0] = invTanHalfFovY / aspect;
399 persp.m[1][1] = invTanHalfFovY;
400 persp.m[2][2] = -(far + near) / (far - near);
401 persp.m[2][3] = -(2 * far * near) / (far - near);
402 persp.m[3][2] = -1.0;
403 persp.m[3][3] = 0.0;
